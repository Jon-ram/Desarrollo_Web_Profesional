    <script>
        // Productos cargados desde JSON
        let productos = [];
        let categoriasUnicas = [];
        let tallasUnicas = [];
        
        // Filtros actuales
        let filtrosActuales = {
            categorias: [],
            tallas: [],
            precioMax: 1000,
            busqueda: "",
            orden: "relevance",
            paginaActual: 1,
            productosPorPagina: 12
        };

        // Elementos DOM
        const elementos = {
            productsGrid: document.getElementById('productsGrid'),
            searchInput: document.getElementById('searchInput'),
            priceRange: document.getElementById('priceRange'),
            priceValue: document.getElementById('priceValue'),
            sortSelect: document.getElementById('sortSelect'),
            clearFiltersBtn: document.getElementById('clearFilters'),
            productCount: document.getElementById('productCount'),
            categoryFilterContainer: document.getElementById('categoryFilterContainer'),
            sizeFilterContainer: document.getElementById('sizeFilterContainer'),
            paginationContainer: document.getElementById('paginationContainer'),
            prevPageBtn: document.getElementById('prevPage'),
            nextPageBtn: document.getElementById('nextPage')
        };

        // Inicializar aplicación
        async function inicializarAplicacion() {
            try {
                // Cargar productos desde JSON
                await cargarProductos();
                
                // Configurar filtros
                configurarFiltros();
                
                // Configurar eventos
                configurarEventos();
                
                // Renderizar productos iniciales
                renderizarProductosFiltrados();
                
                // Configurar modo oscuro
                configurarModoOscuro();
                
            } catch (error) {
                console.error('Error al inicializar:', error);
                mostrarError();
            }
        }

        // Cargar productos desde JSON
        async function cargarProductos() {
            try {
                const respuesta = await fetch('productos.json');
                const datos = await respuesta.json();
                
                productos = datos.productos;
                
                // Extraer categorías únicas
                categoriasUnicas = [...new Set(productos.map(p => p.categoria))];
                
                // Extraer tallas únicas
                const todasTallas = productos.flatMap(p => p.tallas);
                tallasUnicas = [...new Set(todasTallas)].sort();
                
                // Actualizar contador
                elementos.productCount.textContent = `Mostrando ${productos.length} productos`;
                
            } catch (error) {
                console.error('Error al cargar productos:', error);
                throw error;
            }
        }

        // Configurar filtros dinámicos
        function configurarFiltros() {
            // Configurar filtro de categorías
            elementos.categoryFilterContainer.innerHTML = categoriasUnicas.map(categoria => `
                <label class="flex items-center text-sm cursor-pointer group">
                    <input type="checkbox" value="${categoria}" class="hidden category-checkbox"/>
                    <span class="w-4 h-4 border border-slate-300 dark:border-slate-600 rounded mr-3 flex items-center justify-center">
                        <span class="check-icon hidden text-white">✓</span>
                    </span>
                    <span class="group-hover:text-primary transition-colors">${capitalizarPrimeraLetra(categoria)}</span>
                </label>
            `).join('');

            // Configurar filtro de tallas
            elementos.sizeFilterContainer.innerHTML = tallasUnicas.map(talla => `
                <button class="size-filter border border-slate-200 dark:border-slate-700 py-2 text-xs hover:border-primary transition-colors rounded" data-size="${talla.toLowerCase()}">
                    ${talla.toUpperCase()}
                </button>
            `).join('');
        }

        // Configurar eventos
        function configurarEventos() {
            // Evento de búsqueda
            elementos.searchInput.addEventListener('input', function() {
                filtrosActuales.busqueda = this.value.toLowerCase();
                filtrosActuales.paginaActual = 1;
                renderizarProductosFiltrados();
            });

            // Evento de filtro de precio
            elementos.priceRange.addEventListener('input', function() {
                const valor = parseInt(this.value);
                elementos.priceValue.textContent = `$${valor.toLocaleString()}`;
                filtrosActuales.precioMax = valor;
                filtrosActuales.paginaActual = 1;
                renderizarProductosFiltrados();
            });

            // Evento de ordenación
            elementos.sortSelect.addEventListener('change', function() {
                filtrosActuales.orden = this.value;
                renderizarProductosFiltrados();
            });

            // Evento de categorías
            document.querySelectorAll('.category-checkbox').forEach(checkbox => {
                checkbox.addEventListener('change', function() {
                    const categoria = this.value;
                    const checkIcon = this.parentElement.querySelector('.check-icon');
                    
                    if (this.checked) {
                        checkIcon.classList.remove('hidden');
                        checkIcon.parentElement.classList.add('bg-primary', 'border-primary');
                        filtrosActuales.categorias.push(categoria);
                    } else {
                        checkIcon.classList.add('hidden');
                        checkIcon.parentElement.classList.remove('bg-primary', 'border-primary');
                        filtrosActuales.categorias = filtrosActuales.categorias.filter(c => c !== categoria);
                    }
                    
                    filtrosActuales.paginaActual = 1;
                    renderizarProductosFiltrados();
                });
            });

            // Evento de tallas
            document.querySelectorAll('.size-filter').forEach(button => {
                button.addEventListener('click', function() {
                    const talla = this.dataset.size;
                    const estaActivo = this.classList.contains('active');
                    
                    if (estaActivo) {
                        this.classList.remove('active', 'bg-primary', 'text-white', 'border-primary');
                        filtrosActuales.tallas = filtrosActuales.tallas.filter(t => t !== talla);
                    } else {
                        this.classList.add('active', 'bg-primary', 'text-white', 'border-primary');
                        filtrosActuales.tallas.push(talla);
                    }
                    
                    filtrosActuales.paginaActual = 1;
                    renderizarProductosFiltrados();
                });
            });

            // Evento limpiar filtros
            elementos.clearFiltersBtn.addEventListener('click', function() {
                // Limpiar checkboxes de categorías
                document.querySelectorAll('.category-checkbox').forEach(cb => {
                    cb.checked = false;
                    cb.parentElement.querySelector('.check-icon').classList.add('hidden');
                    cb.parentElement.querySelector('span:first-child').classList.remove('bg-primary', 'border-primary');

                elementos.priceValue.textContent = '$1,000';

                // Limpiar ordenación
                elementos.sortSelect.value = 'relevance';
                
                // Restablecer filtros
                filtrosActuales = {
                    categorias: [],
                    tallas: [],
                    precioMax: 1000,
                    busqueda: "",
                    orden: "relevance",
                    paginaActual: 1,
                    productosPorPagina: 12
                };
                
                renderizarProductosFiltrados();
            });
        }

        // Filtrar productos
        function filtrarProductos() {
            let productosFiltrados = [...productos];
            
            // Filtrar por búsqueda
            if (filtrosActuales.busqueda) {
                productosFiltrados = productosFiltrados.filter(producto =>
                    producto.nombre.toLowerCase().includes(filtrosActuales.busqueda) ||
                    producto.descripcion.toLowerCase().includes(filtrosActuales.busqueda) ||
                    producto.tags.some(tag => tag.includes(filtrosActuales.busqueda))
                );
            }
            
            // Filtrar por categorías
            if (filtrosActuales.categorias.length > 0) {
                productosFiltrados = productosFiltrados.filter(producto =>
                    filtrosActuales.categorias.includes(producto.categoria)
                );
            }
            
            // Filtrar por tallas
            if (filtrosActuales.tallas.length > 0) {
                productosFiltrados = productosFiltrados.filter(producto =>
                    producto.tallas.some(talla =>
                        filtrosActuales.tallas.includes(talla.toLowerCase())
                    )
                );
            }
            
            // Filtrar por precio
            productosFiltrados = productosFiltrados.filter(producto =>
                producto.precio <= filtrosActuales.precioMax
            );
            
            // Ordenar productos
            productosFiltrados.sort((a, b) => {
                switch (filtrosActuales.orden) {
                    case 'price-low':
                        return a.precio - b.precio;
                    case 'price-high':
                        return b.precio - a.precio;
                    case 'newest':
                        return b.id - a.id;
                    default: // relevance
                        // Primero destacados, luego por ID
                        if (a.destacado && !b.destacado) return -1;
                        if (!a.destacado && b.destacado) return 1;
                        return b.id - a.id;
                }
            });
            
            return productosFiltrados;
        }

        // Renderizar productos filtrados
        function renderizarProductosFiltrados() {
            const productosFiltrados = filtrarProductos();
            const totalProductos = productosFiltrados.length;
            
            // Calcular paginación
            const totalPaginas = Math.ceil(totalProductos / filtrosActuales.productosPorPagina);
            const inicio = (filtrosActuales.paginaActual - 1) * filtrosActuales.productosPorPagina;
            const fin = inicio + filtrosActuales.productosPorPagina;
            const productosPagina = productosFiltrados.slice(inicio, fin);
            
            // Actualizar contador
            elementos.productCount.textContent = `Mostrando ${inicio + 1}-${Math.min(fin, totalProductos)} de ${totalProductos} productos`;
            
            // Renderizar productos
            elementos.productsGrid.innerHTML = '';
            
            if (productosPagina.length === 0) {
                elementos.productsGrid.innerHTML = `
                    <div class="col-span-full text-center py-16">
                        <span class="material-symbols-outlined text-6xl text-slate-300 dark:text-slate-700 mb-4">search_off</span>
                        <h3 class="text-xl font-semibold mb-2">No se encontraron productos</h3>
                        <p class="text-slate-500">Intenta con otros filtros o términos de búsqueda.</p>
                    </div>
                `;
            } else {
                productosPagina.forEach(producto => {
                    const productCard = crearTarjetaProducto(producto);
                    elementos.productsGrid.appendChild(productCard);
                });
            }
            
            // Actualizar paginación
            actualizarPaginacion(totalPaginas, totalProductos);
        }

        // Crear tarjeta de producto

            div.addEventListener('click', (e) => {

        // Actualizar paginación
        function actualizarPaginacion(totalPaginas, totalProductos) {
            if (totalProductos > filtrosActuales.productosPorPagina) {
                elementos.paginationContainer.classList.remove('hidden');
                
                // Actualizar estado de botones
                elementos.prevPageBtn.disabled = filtrosActuales.paginaActual === 1;
                elementos.nextPageBtn.disabled = filtrosActuales.paginaActual === totalPaginas;
                
                // Evento página anterior
                elementos.prevPageBtn.onclick = () => {
                    if (filtrosActuales.paginaActual > 1) {
                        filtrosActuales.paginaActual--;
                        renderizarProductosFiltrados();
                    }
                };
                
                // Evento página siguiente
                elementos.nextPageBtn.onclick = () => {
                    if (filtrosActuales.paginaActual < totalPaginas) {
                        filtrosActuales.paginaActual++;
                        renderizarProductosFiltrados();
                    }
                };
                
            } else {
                elementos.paginationContainer.classList.add('hidden');
            }
        }

            `;
            
            document.body.appendChild(notification);
            
            // Remover notificación después de 3 segundos
            setTimeout(() => {
                notification.remove();
            }, 3000);
            
            // Aquí puedes agregar lógica para guardar en el carrito
            console.log('Producto agregado:', producto);
        }

        // Configurar modo oscuro
        function configurarModoOscuro() {
            const themeToggleBtn = document.getElementById('theme-toggle');
            const htmlElement = document.documentElement;
            
            if (localStorage.getItem('theme') === 'dark' || (!('theme' in localStorage) && window.matchMedia('(prefers-color-scheme: dark)').matches)) {
                htmlElement.classList.add('dark');
            } else {
                htmlElement.classList.remove('dark');
            }
            
            themeToggleBtn.addEventListener('click', () => {
                htmlElement.classList.toggle('dark');
                localStorage.setItem('theme', htmlElement.classList.contains('dark') ? 'dark' : 'light');
            });
        }

        // Mostrar error
        function mostrarError() {
            elementos.productsGrid.innerHTML = `
                <div class="col-span-full text-center py-16">

    <script>
        // Estado del producto
        const estadoProducto = {
            id: null,
            datos: null,
            tallaSeleccionada: null,
            cantidad: 1,
            enFavoritos: false
        };

        // Obtener ID del producto desde la URL
        function obtenerIdProducto() {
            const params = new URLSearchParams(window.location.search);
            return parseInt(params.get('id')) || 1; // Default to product 1
        }

        // Cargar datos del producto
        async function cargarProducto() {
            try {
                const productId = obtenerIdProducto();
                
                // Cargar todos los productos
                const respuesta = await fetch('productos.json');
                const datos = await respuesta.json();
                
                // Encontrar el producto específico
                const producto = datos.productos.find(p => p.id === productId);
                
                if (!producto) {
                    throw new Error('Producto no encontrado');
                }
                
                estadoProducto.id = productId;
                estadoProducto.datos = producto;
                
                // Renderizar el producto
                renderizarProducto(producto);
                
                // Cargar productos relacionados
                cargarProductosRelacionados(producto, datos.productos);
                
                // Ocultar loading, mostrar contenido
                document.getElementById('loadingState').classList.add('hidden');
                document.getElementById('productContainer').classList.remove('hidden');
                
            } catch (error) {
                console.error('Error al cargar el producto:', error);
                mostrarError();
            }
        }

        // Renderizar producto
        function renderizarProducto(producto) {
            // Clonar template
            const template = document.getElementById('productTemplate');
            const contenido = template.content.cloneNode(true);
            
            // Actualizar breadcrumb
            const breadcrumb = document.getElementById('breadcrumb');
            const breadcrumbText = breadcrumb.querySelector('span:last-child');
            breadcrumbText.textContent = producto.nombre;
            
            // Actualizar título de página
            document.title = `${producto.nombre} | ATELIER`;
            
            // Actualizar contenido principal
            const container = document.getElementById('productContainer');
            container.innerHTML = '';
            container.appendChild(contenido);
            
            // Configurar elementos del producto
            configurarElementosProducto(producto);
            
            // Configurar eventos
            configurarEventosProducto();
        }

        // Configurar elementos del producto
        function configurarElementosProducto(producto) {
            // Información básica
            document.getElementById('productCategory').textContent = capitalizarPrimeraLetra(producto.categoria);
            document.getElementById('productCode').textContent = `• Código: ATL-${producto.id.toString().padStart(3, '0')}`;
            document.getElementById('productName').textContent = producto.nombre;
            document.getElementById('productDescription').textContent = producto.descripcion;
            
            // Precios
            const currentPrice = document.getElementById('currentPrice');
            const originalPrice = document.getElementById('originalPrice');
            const discountBadge = document.getElementById('discountBadge');
            
            currentPrice.textContent = `$${producto.precio}`;
            
            if (producto.en_oferta && producto.precio_original) {
                originalPrice.textContent = `$${producto.precio_original}`;
                const descuento = Math.round((1 - producto.precio / producto.precio_original) * 100);
                discountBadge.textContent = `-${descuento}%`;
                discountBadge.classList.remove('hidden');
            } else {
                originalPrice.classList.add('hidden');
                discountBadge.classList.add('hidden');
            }
            
            // Valoración
            const ratingStars = document.getElementById('ratingStars');
            const ratingText = document.getElementById('ratingText');
            const averageRating = document.getElementById('averageRating');
            const reviewStars = document.getElementById('reviewStars');
            const reviewCount = document.getElementById('reviewCount');
            
            const valoracion = producto.valoracion || 4.2;
            const reseñas = producto.reseñas || 128;
            
            // Generar estrellas para rating
            ratingStars.innerHTML = generarEstrellasHTML(valoracion);
            ratingText.textContent = `${valoracion} (${reseñas} reseñas)`;
            
            // Generar estrellas para sección de reseñas
            averageRating.textContent = valoracion;
            reviewStars.innerHTML = generarEstrellasHTML(valoracion);
            reviewCount.textContent = `Basado en ${reseñas} reseñas`;
            
            // Stock
            const stockStatus = document.getElementById('stockStatus');
            const stockInfo = document.getElementById('stockInfo');
            
            if (producto.stock > 0) {
                stockStatus.textContent = 'En stock';
                stockInfo.innerHTML = `Solo quedan <span class="font-bold text-primary">${producto.stock}</span> unidades`;
                document.getElementById('quantity').max = Math.min(producto.stock, 10);
            } else {
                stockStatus.textContent = 'Agotado';
                stockStatus.classList.add('text-red-500');
                stockInfo.textContent = 'Producto agotado';
                document.getElementById('addToCartBtn').disabled = true;
                document.getElementById('addToCartBtn').classList.add('opacity-50', 'cursor-not-allowed');
                document.getElementById('quantity').disabled = true;
            }
            
            // Imágenes
            const mainImage = document.getElementById('mainImage');
            const thumbnailsContainer = document.getElementById('thumbnailsContainer');
            const badgeContainer = document.getElementById('badgeContainer');
            
            // Imagen principal
            mainImage.src = producto.imagen;
            mainImage.alt = producto.nombre;
            
            // Badges
            badgeContainer.innerHTML = '';
            if (producto.en_oferta) {
                badgeContainer.innerHTML += `
                    <span class="absolute top-4 left-4 bg-primary text-white text-xs font-bold uppercase tracking-widest px-3 py-1.5 rounded-lg">
                        ¡OFERTA ESPECIAL!
                    </span>
                `;
            }
            
            if (producto.nuevo) {
                badgeContainer.innerHTML += `
                    <span class="absolute top-16 left-4 bg-green-500 text-white text-xs font-bold uppercase tracking-widest px-3 py-1.5 rounded-lg">
                        NUEVO
                    </span>
                `;
            }
            
            // Miniaturas
            thumbnailsContainer.innerHTML = '';
            
            // Miniatura para imagen principal
            const miniaturaPrincipal = document.createElement('button');
            miniaturaPrincipal.className = 'thumbnail-btn border-2 border-primary rounded-lg overflow-hidden active';
            miniaturaPrincipal.innerHTML = `<img src="${producto.imagen}" alt="${producto.nombre}" class="w-full aspect-square object-cover"/>`;
            miniaturaPrincipal.addEventListener('click', () => cambiarImagenPrincipal(producto.imagen, miniaturaPrincipal));
            thumbnailsContainer.appendChild(miniaturaPrincipal);
            
            // Miniaturas para imágenes secundarias
            if (producto.imagenes_secundarias && producto.imagenes_secundarias.length > 0) {

        // Cargar productos relacionados
        async function cargarProductosRelacionados(productoActual, todosProductos) {
            const relatedProductsContainer = document.getElementById('relatedProducts');
            
            if (!relatedProductsContainer) return;
            
            // Filtrar productos relacionados (misma categoría, excluyendo el actual)
            const productosRelacionados = todosProductos
                .filter(p => p.id !== productoActual.id && p.categoria === productoActual.categoria)
                .slice(0, 4); // Tomar máximo 4 productos
            
            if (productosRelacionados.length === 0) {
                // Si no hay productos en la misma categoría, tomar algunos aleatorios
                const otrosProductos = todosProductos
                    .filter(p => p.id !== productoActual.id)
                    .slice(0, 4);
                renderizarProductosRelacionados(otrosProductos);
            } else {
                renderizarProductosRelacionados(productosRelacionados);
            }
        }

        // Renderizar productos relacionados
        function renderizarProductosRelacionados(productos) {
            const container = document.getElementById('relatedProducts');
            container.innerHTML = '';
            
            productos.forEach(producto => {
                const productCard = document.createElement('a');
                productCard.href = `producto.html?id=${producto.id}`;
                productCard.className = 'group block';
                
                const precioOriginalHTML = producto.en_oferta && producto.precio_original ?
                    `<span class="text-slate-400 line-through text-sm">$${producto.precio_original}</span>` : '';
                
                const precioHTML = producto.en_oferta ?
                    `<div class="flex items-center justify-center gap-2">
                        <span class="text-primary font-bold text-lg">$${producto.precio}</span>
                        ${precioOriginalHTML}
                    </div>` :
                    `<p class="text-slate-900 dark:text-white font-bold text-lg">$${producto.precio}</p>`;
                
                productCard.innerHTML = `
                    <div class="relative overflow-hidden rounded-lg aspect-[3/4] mb-3 bg-slate-100 dark:bg-slate-800">
                        <img src="${producto.imagen}" alt="${producto.nombre}" class="w-full h-full object-cover group-hover:scale-105 transition-transform duration-300"/>
                    </div>
                    <div class="text-center">
                        <h3 class="font-medium text-base mb-1">${producto.nombre}</h3>
                        ${precioHTML}
                    </div>
                `;
                
                container.appendChild(productCard);
            });
        }

        // Guardar en carrito (simulación)
        function guardarEnCarrito(item) {
            // Obtener carrito actual de localStorage
            let carrito = JSON.parse(localStorage.getItem('carritoAtelier')) || [];
            
            // Verificar si el producto ya está en el carrito
            const itemIndex = carrito.findIndex(i => 
                i.id === item.id && i.talla === item.talla
            );
            
            if (itemIndex > -1) {
                // Actualizar cantidad
                carrito[itemIndex].cantidad += item.cantidad;
            } else {
                // Añadir nuevo item
                carrito.push(item);
            }
            
            // Guardar en localStorage
            localStorage.setItem('carritoAtelier', JSON.stringify(carrito));
            
            // Actualizar contador del carrito si existe
            const cartCount = document.querySelector('.cart-count');
            if (cartCount) {
                const totalItems = carrito.reduce((sum, item) => sum + item.cantidad, 0);
                cartCount.textContent = totalItems;
                cartCount.classList.remove('hidden');
            }
        }

        // Generar HTML de estrellas
        function generarEstrellasHTML(rating) {
            const estrellasLlenas = Math.floor(rating);
            const tieneMediaEstrella = rating % 1 >= 0.5;
            const estrellasVacias = 5 - estrellasLlenas - (tieneMediaEstrella ? 1 : 0);
            
            let html = '';
            
            // Estrellas llenas
            for (let i = 0; i < estrellasLlenas; i++) {
                html += '<span class="material-symbols-outlined text-amber-500">star</span>';
            }
            
            // Media estrella
            if (tieneMediaEstrella) {
                html += '<span class="material-symbols-outlined text-amber-500">star_half</span>';
            }
            
            // Estrellas vacías
            for (let i = 0; i < estrellasVacias; i++) {
                html += '<span class="material-symbols-outlined text-slate-300">star</span>';
            }
            
            return html;
        }
